!
! SW-F1 (Cisco 2960-24TT) - First Floor Access Switch 1
!
hostname SW-F1
!
! Create VLANs
vlan 10
 name STAFF
exit
vlan 20
 name ADMIN
exit
vlan 30
 name GUEST
exit
vlan 40
 name NETWORK_DEV
exit
vlan 50
 name VOIP
exit
vlan 60
 name PRINTERS
exit
vlan 70
 name AP-MGMT
exit
vlan 99
 name IOT
exit
!
! Access ports for Staff PCs (VLAN 10)
interface range fa0/1-20
 switchport mode access
 switchport access vlan 10
 switchport voice vlan 50
 spanning-tree portfast
 no shutdown
exit
!
! Access ports for Admin PCs (VLAN 20)
interface range fa0/21-22
 switchport mode access
 switchport access vlan 20
 spanning-tree portfast
 no shutdown
exit
!
! Access ports for IP Phones (VLAN 50)
interface range fa0/23-24
 switchport mode access
 switchport access vlan 50
 switchport voice vlan 50
 spanning-tree portfast
 no shutdown
exit
!
! Trunk to L3-CORE
interface gig0/1
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30,40,50,60,70,99
 no shutdown
exit
!
! Trunk to SW-F2
interface gig0/2
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30,40,50,60,70,99
 no shutdown
exit
!
! Port Security
interface range fa0/1-24
 switchport port-security
 switchport port-security maximum 2
 switchport port-security violation restrict
exit
!
! DHCP Snooping
ip dhcp snooping
ip dhcp snooping vlan 10,20,30,50,60,70,99
interface gig0/1
 ip dhcp snooping trust
interface gig0/2
 ip dhcp snooping trust
exit
!
! Save configuration
write memory
