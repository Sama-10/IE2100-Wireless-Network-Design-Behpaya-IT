!
! NGFW (Cisco 2911 Router) - Next Generation Firewall
! NAT, ACLs, VPN
!
hostname NGFW
!
! Outside interface (to ISP Router)
interface gig0/0
 ip address 209.165.200.2 255.255.255.248
 ip nat outside
 no shutdown
exit
!
! Inside interface (to L3-CORE)
interface gig0/1
 ip address 10.0.0.1 255.255.255.252
 ip nat inside
 no shutdown
exit
!
! NAT ACL - Internal networks
access-list 1 permit 192.168.0.0 0.0.255.255
access-list 1 permit 172.16.0.0 0.0.255.255
!
! Apply NAT
ip nat inside source list 1 interface gig0/0 overload
!
! Default route to ISP
ip route 0.0.0.0 0.0.0.0 209.165.200.1
!
! Firewall ACL - Block Guest from internal networks
access-list 101 deny ip 172.16.30.0 0.0.0.255 192.168.0.0 0.0.255.255
access-list 101 permit ip any any
!
! Apply ACL to inside interface
interface gig0/1
 ip access-group 101 in
exit
!
! Save configuration
write memory
